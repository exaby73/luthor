```dart
import 'package:luthor/luthor.dart';
import 'package:freezed_annotation/freezed_annotation.dart';

part 'email.freezed.dart';
part 'email.g.dart';

@luthor
@freezed
abstract class EmailSchema with _$EmailSchema {
  const factory EmailSchema({
    @isEmail /* OR @IsEmail() */ required String value,
  }) = _EmailSchema;

  factory EmailSchema.fromJson(Map<String, dynamic> json) =>
      _$EmailSchemaFromJson(json);
}

void main() {
  final result = $EmailSchemaValidate({'value': 'user@example.com'});

  switch (result) {
    case SchemaValidationSuccess(data: final data):
      print('✅ Valid: ${data.value}');
    case SchemaValidationError(errors: final errors):
      print('❌ Errors: $errors');
  }
}
```
