```dart
import 'package:luthor/luthor.dart';
import 'package:freezed_annotation/freezed_annotation.dart';

part 'length.freezed.dart';
part 'length.g.dart';

@freezed
@luthor
class LengthSchema extends _$LengthSchema {
  const factory LengthSchema({
    required String value,
  }) = _$LengthSchema;

  static SchemaValidationResult<LengthSchema> validate(
    Map<String, dynamic> json,
  ) =>
      _$LengthSchemaValidate(json);
  
  factory LengthSchema.fromJson(Map<String, dynamic> json) =>
      _$LengthSchemaFromJson(json);
}

void main() {
  print(LengthSchema.validate({'value': 'Hello'}));
}
```
