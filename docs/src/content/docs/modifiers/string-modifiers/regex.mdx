```dart
import 'package:luthor/luthor.dart';
import 'package:freezed_annotation/freezed_annotation.dart';

part 'regex.freezed.dart';
part 'regex.g.dart';

@freezed
@luthor
class RegexSchema extends _$RegexSchema {
  const factory RegexSchema({
    required String value,
  }) = _$RegexSchema;

  static SchemaValidationResult<RegexSchema> validate(
    Map<String, dynamic> json,
  ) =>
      _$RegexSchemaValidate(json);
  
  factory RegexSchema.fromJson(Map<String, dynamic> json) =>
      _$RegexSchemaFromJson(json);
}

void main() {
  print(RegexSchema.validate({'value': 'Hello123'}));
}
```
