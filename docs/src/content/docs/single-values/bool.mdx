import 'package:luthor/luthor.dart';
import 'package:freezed_annotation/freezed_annotation.dart';

part 'bool_schema.freezed.dart';
part 'bool_schema.g.dart';

@freezed
@luthor
class BoolSchema extends _$BoolSchema {
  const factory BoolSchema({
    required bool value,
  }) = _$BoolSchema;

  static SchemaValidationResult<BoolSchema> validate(
    Map<String, dynamic> json,
  ) =>
      _$BoolSchemaValidate(json);
}

void main() {
  print(BoolSchema.validate({'value': true}));
}
