import 'package:luthor/luthor.dart';
import 'package:freezed_annotation/freezed_annotation.dart';

part 'any_schema.freezed.dart';
part 'any_schema.g.dart';

@freezed
@luthor
class AnySchema extends _$AnySchema {
  const factory AnySchema({
    required dynamic value,
  }) = _$AnySchema;

  static SchemaValidationResult<AnySchema> validate(
    Map<String, dynamic> json,
  ) =>
      _$AnySchemaValidate(json);
}

void main() {
  print(AnySchema.validate({'value': 42}));
}
